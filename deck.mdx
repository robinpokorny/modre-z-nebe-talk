```ts
type IncomingOrder = { items: string[] };
type AcceptedOrder = { items: string[]; acceptedBy: string };

type Order = IncomingOrder | AcceptedOrder;

type processOrder = (order: Order) => AcceptedOrder;
type fulfillOrder = (order: AcceptedOrder) => Shipment;
```
